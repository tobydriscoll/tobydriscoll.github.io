The 5-body problem

[f,tspan,u0] = Nbody(5);  % problem setup

% The popular RK45 method
[t,u] = ode45( f, tspan, u0 );
size(t)  % time points
size(u)  % solution history
plot(t,u,'.-')  % note nonuniform time steps

ans =

   129     1


ans =

   129    20
