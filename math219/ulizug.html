def is_outlier(x):
    Q25, Q75 = x.quantile([.25,.75])
    I = Q75 - Q25
    return ((x < (Q25 - 1.5*I)) |  (x > (Q75 + 1.5*I)))
    
outliers = watq.groupby("Site_Id")['Salinity (ppt)'].transform(is_outlier)
q = watq.loc[outliers, 'Site_Id'].value_counts()

print(q)
